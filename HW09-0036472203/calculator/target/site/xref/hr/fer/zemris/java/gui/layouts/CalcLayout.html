11  /**
12   * <p>
13   * Layout suitable for graphic representation of a calculator. A grid-like
14   * layout consists of 5 rows and 7 columns. Position for first component - (1,
15   * 1) - has the largest width and it extends through positions (1, 1), (1, 2),
16   * (1, 3), (1, 4) and (1, 5). Every other component occupies one position.
17   * Spacing between positions can be provided. If not, spacing is zero.
18   * </p>

69  	/**
70  	 * Enum used for determination which of {@code getMinimumSize},
71  	 * {@code getPreferredSize} or {@code getMaximumSize} methods to use when
72  	 * searching for component size.
73  	 * 
74  	 * @author Petra Rebernjak - 0036472203
75  	 *
76  	 */
77  	private static enum Size {
78  		/**
79  		 * Used when {@code getMinimumSize} method needs to be invoked on
80  		 * component.
81  		 */
82  		MINIMUM,
83  		/**
84  		 * Used when {@code getPreferredSize} method needs to be invoked on
85  		 * component.
86  		 */
87  		PREFERRED,
88  		/**
89  		 * Used when {@code getMaximumSize} method needs to be invoked on
90  		 * component.
91  		 */
92  		MAXIMUM;
93  
94  		/**
95  		 * Returns predefined size of the component.
96  		 * 
97  		 * @param comp
98  		 *            Component to calculate size of.
99  		 * @return Predefined size of the component.
100 		 */
101 		Dimension getPredefinedSize(Component comp) {
102 			Dimension d = null;
103 			if (this.equals(Size.MINIMUM)) {
104 				d = comp.getMinimumSize();
105 			} else if (this.equals(Size.PREFERRED)) {
106 				d = comp.getPreferredSize();
107 			} else if (this.equals(Size.MAXIMUM)) {
108 				d = comp.getMaximumSize();
109 			}
110 			return d;
111 		}
112 	}

188 	@Override
189 	public void layoutContainer(Container parent) {
190 		Insets insets = parent.getInsets();
191 		int width = (parent.getWidth() + subWidthInsets(insets) - getSpacingWidth())
192 				/ NUM_COLS;
193 		int height = (parent.getHeight() + subHeightInsets(insets) - getSpacingHeight())
194 				/ NUM_ROWS;
195 		for (Component component : parent.getComponents()) {
196 			RCPosition position = components.get(component);
197 			int x = insets.left + width * (position.getColumn() - 1) + space
198 					* position.getColumn();
199 			int y = insets.top + height * (position.getRow() - 1) + space
200 					* position.getRow();
201 
202 			int properWidth = width;
203 			if (position.equals(FIRS_POSITION)) {
204 				properWidth = width * NUM_COLS_FIRST + space
205 						* (NUM_COLS_FIRST - 1);
206 			}
207 
208 			component.setBounds(x, y, properWidth, height);
209 		}
210 
211 	}

213 	@Override
214 	public void addLayoutComponent(Component comp, Object constr) {
215 		RCPosition constraints = null;
216 
217 		if (constr instanceof String) {
218 			constraints = RCPosition.parse((String) constr);
219 		} else if (constr instanceof RCPosition) {
220 			constraints = (RCPosition) constr;
221 		}
222 
223 		int row = constraints.getRow();
224 		int column = constraints.getColumn();
225 
226 		// If constraint with at least one negative index throw exception.
227 		if (row < 1 || row > NUM_ROWS || column < 1 || column > NUM_COLS) {
228 			throw new IllegalArgumentException(
229 					"Given positions cannot be zero or negative: "
230 							+ constraints.toString());
231 		}
232 
233 		// If constraint with indexes (1, 2) - (1, 5) throw exception.
234 		if (row == 1 && column > 1 && column < 6) {
235 			throw new IllegalArgumentException(
236 					"Given positions cannot be (1, 2) - (1, 5): "
237 							+ constraints.toString());
238 		}
239 
240 		// If constraint with indexes that already exists throw exception.
241 		if (components.containsValue(constraints)) {
242 			throw new IllegalArgumentException("Given position already full: "
243 					+ constraints.toString());
244 		}
245 
246 		// Add to map.
247 		components.put(comp, constraints);
248 	}

270 	/**
271 	 * Calculates the layout size based on components size.
272 	 * 
273 	 * @param parent
274 	 *            Parent container.
275 	 * @param size
276 	 *            Enum for determining what size to calculate - minimum,
277 	 *            preferred or maximum.
278 	 * @return Size of the layout.
279 	 */
280 	private Dimension getLayoutSize(Container parent, Size size) {
281 		int width = 0;
282 		int height = 0;
283 
284 		int numberOfComponents = parent.getComponentCount();
285 		for (int i = 0; i < numberOfComponents; i++) {
286 			Component component = parent.getComponent(i);
287 			Dimension d = size.getPredefinedSize(component);
288 			if (components.get(component).equals(FIRS_POSITION)) {
289 				d.width = (component.getWidth() - (NUM_COLS_FIRST - 1) * space)
290 						/ NUM_COLS_FIRST;
291 			}
292 			if (i > 0) {
293 				width = Math.max(width, (int) d.getWidth());
294 				height = Math.max(height, (int) d.getHeight());
295 			} else {
296 				width = d.width;
297 				height = d.height;
298 			}
299 		}
300 
301 		Insets insets = parent.getInsets();
302 		width = width * NUM_COLS + addWidthInsets(insets) + getSpacingWidth();
303 		height = height * NUM_ROWS + addHeightInsets(insets)
304 				+ getSpacingHeight();
305 
306 		return new Dimension(width, height);
307 	}
308 
309 	private int addWidthInsets(Insets insets) {
310 		return insets.left + insets.right;
311 	}
312 
313 	private int subWidthInsets(Insets insets) {
314 		return -addWidthInsets(insets);
315 	}
316 
317 	private int addHeightInsets(Insets insets) {
318 		return insets.top + insets.bottom;
319 	}
320 
321 	private int subHeightInsets(Insets insets) {
322 		return -addHeightInsets(insets);
323 	}
324 
325 	/**
326 	 * Returns width of all the spacing between columns.
327 	 * 
328 	 * @return Width of all the spacing between columns.
329 	 */
330 	private int getSpacingWidth() {
331 		int spacingWidth = (NUM_COLS + 1) * space;
332 		return (spacingWidth) == 0 ? 0 : spacingWidth;
333 	}
334 
335 	/**
336 	 * Returns height of all the spacing between rows.
337 	 * 
338 	 * @return Height of all the spacing between rows.
339 	 */
340 	private int getSpacingHeight() {
341 		int spacingHeight = (NUM_ROWS + 1) * space;
342 		return (spacingHeight) == 0 ? 0 : spacingHeight;
343 	}
